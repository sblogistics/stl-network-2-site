# Python example using Selenium for USS portal
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverConnect
from selenium.webdriver.support import expected_conditions as EC
import time
import json

class USSAuctionScraper:
    def __init__(self, username, password):
        self.username = username
        self.password = password
        self.driver = webdriver.Chrome()  # or Firefox
        
    def login(self):
        """Login to USS member portal"""
        self.driver.get("https://uss-member-portal-url.jp/login")
        
        # Enter credentials
        username_field = self.driver.find_element(By.ID, "username")
        password_field = self.driver.find_element(By.ID, "password")
        
        username_field.send_keys(self.username)
        password_field.send_keys(self.password)
        
        # Submit login
        login_button = self.driver.find_element(By.ID, "login-button")
        login_button.click()
        
        time.sleep(3)  # Wait for login
        
    def search_vehicles(self, make=None, model=None, year_from=None):
        """Search for vehicles"""
        # Navigate to search page
        self.driver.get("https://uss-member-portal-url.jp/search")
        
        # Fill search form
        if make:
            make_field = self.driver.find_element(By.NAME, "make")
            make_field.send_keys(make)
            
        if model:
            model_field = self.driver.find_element(By.NAME, "model")
            model_field.send_keys(model)
            
        if year_from:
            year_field = self.driver.find_element(By.NAME, "year_from")
            year_field.send_keys(str(year_from))
        
        # Submit search
        search_button = self.driver.find_element(By.ID, "search-button")
        search_button.click()
        
        time.sleep(2)  # Wait for results
        
        # Parse results
        return self.parse_results()
        
    def parse_results(self):
        """Parse vehicle listing results"""
        vehicles = []
        
        # Find all vehicle cards (adjust selectors based on actual USS portal)
        vehicle_elements = self.driver.find_elements(By.CLASS_NAME, "vehicle-item")
        
        for element in vehicle_elements:
            try:
                vehicle = {
                    'lot_number': element.find_element(By.CLASS_NAME, "lot-number").text,
                    'make': element.find_element(By.CLASS_NAME, "make").text,
                    'model': element.find_element(By.CLASS_NAME, "model").text,
                    'year': element.find_element(By.CLASS_NAME, "year").text,
                    'mileage': element.find_element(By.CLASS_NAME, "mileage").text,
                    'grade': element.find_element(By.CLASS_NAME, "grade").text,
                    'start_price': element.find_element(By.CLASS_NAME, "price").text,
                    'auction_date': element.find_element(By.CLASS_NAME, "auction-date").text,
                    'image_url': element.find_element(By.TAG_NAME, "img").get_attribute("src")
                }
                vehicles.append(vehicle)
            except Exception as e:
                print(f"Error parsing vehicle: {e}")
                continue
                
        return vehicles
        
    def get_vehicle_details(self, lot_number):
        """Get detailed info for specific vehicle"""
        self.driver.get(f"https://uss-member-portal-url.jp/vehicle/{lot_number}")
        
        time.sleep(2)
        
        details = {
            'lot_number': lot_number,
            'chassis_number': self.driver.find_element(By.ID, "chassis").text,
            'engine_size': self.driver.find_element(By.ID, "engine").text,
            'transmission': self.driver.find_element(By.ID, "transmission").text,
            'color': self.driver.find_element(By.ID, "color").text,
            'equipment': self.driver.find_element(By.ID, "equipment").text,
            'inspection_details': self.driver.find_element(By.ID, "inspection").text,
            'photos': [img.get_attribute("src") for img in self.driver.find_elements(By.CLASS_NAME, "vehicle-photo")]
        }
        
        return details
        
    def close(self):
        """Close browser"""
        self.driver.quit()

// Search vehicles
app.get('/api/vehicles/search', async (req, res) => {
    try {
        const { make, model, year_from, year_to, price_max, auction_house, grade } = req.query;
        
        let query = 'SELECT * FROM vehicles WHERE status = "upcoming"';
        const params = [];
        
        if (make) {
            query += ' AND make = ?';
            params.push(make);
        }
        
        if (model) {
            query += ' AND model LIKE ?';
            params.push(`%${model}%`);
        }
        
        if (year_from) {
            query += ' AND year >= ?';
            params.push(parseInt(year_from));
        }
        
        if (year_to) {
            query += ' AND year <= ?';
            params.push(parseInt(year_to));
        }
        
        if (price_max) {
            query += ' AND start_price <= ?';
            params.push(parseFloat(price_max));
        }
        
        if (auction_house) {
            query += ' AND auction_house = ?';
            params.push(auction_house);
        }
        
        if (grade) {
            query += ' AND grade >= ?';
            params.push(grade);
        }
        
        query += ' ORDER BY auction_date ASC, lot_number ASC LIMIT 100';
        
        const [rows] = await pool.query(query, params);
        
        res.json({
            success: true,
            count: rows.length,
            vehicles: rows
        });
    } catch (error) {
        console.error('Search error:', error);
        res.status(500).json({ success: false, error: 'Search failed' });
    }
});

// Get vehicle details
app.get('/api/vehicles/:id', async (req, res) => {
    try {
        const { id } = req.params;
        
        const [rows] = await pool.query(
            'SELECT * FROM vehicles WHERE id = ?',
            [id]
        );
        
        if (rows.length === 0) {
            return res.status(404).json({ success: false, error: 'Vehicle not found' });
        }
        
        res.json({
            success: true,
            vehicle: rows[0]
        });
    } catch (error) {
        console.error('Details error:', error);
        res.status(500).json({ success: false, error: 'Failed to fetch details' });
    }
});

// Get statistics
app.get('/api/stats', async (req, res) => {
    try {
        const [totalVehicles] = await pool.query(
            'SELECT COUNT(*) as count FROM vehicles WHERE status = "upcoming"'
        );
        
        const [makeDistribution] = await pool.query(
            'SELECT make, COUNT(*) as count FROM vehicles WHERE status = "upcoming" GROUP BY make ORDER BY count DESC LIMIT 10'
        );
        
        const [avgPrices] = await pool.query(
            'SELECT make, AVG(start_price) as avg_price FROM vehicles WHERE status = "upcoming" GROUP BY make ORDER BY avg_price DESC LIMIT 10'
        );
        
        res.json({
            success: true,
            total_vehicles: totalVehicles[0].count,
            make_distribution: makeDistribution,
            avg_prices: avgPrices
        });
    } catch (error) {
        console.error('Stats error:', error);
        res.status(500).json({ success: false, error: 'Failed to fetch stats' });
    }
});

// Create customer alert
app.post('/api/alerts', async (req, res) => {
    try {
        const { email, make, model, year_from, year_to, price_max } = req.body;
        
        await pool.query(
            'INSERT INTO customer_alerts (customer_email, make, model, year_from, year_to, price_max) VALUES (?, ?, ?, ?, ?, ?)',
            [email, make, model, year_from, year_to, price_max]
        );
        
        res.json({ success: true, message: 'Alert created successfully' });
    } catch (error) {
        console.error('Alert error:', error);
        res.status(500).json({ success: false, error: 'Failed to create alert' });
    }
});

        function sortResults(sortBy) {
            let sorted = [...currentVehicles];
            
            switch(sortBy) {
                case 'price_asc':
                    sorted.sort((a, b) => (a.start_price || 0) - (b.start_price || 0));
                    break;
                case 'price_desc':
                    sorted.sort((a, b) => (b.start_price || 0) - (a.start_price || 0));
                    break;
                case 'year_desc':
                    sorted.sort((a, b) => (b.year || 0) - (a.year || 0));
                    break;
                case 'mileage_asc':
                    sorted.sort((a, b) => (a.mileage || 0) - (b.mileage || 0));
                    break;
                default: // date
                    sorted.sort((a, b) => new Date(a.auction_date) - new Date(b.auction_date));
            }
            
            displayResults(sorted);
        }
        
        async function viewVehicleDetails(vehicleId) {

                const response = await fetch(`${API_URL}/vehicles/${vehicleId}`);
                const data = await response.json();

                    // Open in modal or new page
                    window.location.href = `vehicle-details.html?id=${vehicleId}`;
                }
            } catch (error) {
                console.error('Error:', error);
                alert('Failed to load vehicle details');
            }
        }
